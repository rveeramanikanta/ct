Stack using Arrays

#include <stdio.h>
#define STACK_MAX_SIZE 10
						
int arr[STACK_MAX_SIZE];
int top = -1;
					
void main() {

}
void push(int element) {
	if(top == STACK_MAX_SIZE - 1) {
		printf("Stack is overflow.");
	} else {
		top = top + 1;
		arr[top] = element;
		printf("Successfully pushed.");
	}
}
					
void pop() {
	int x;
	if(top < 0) {
		printf("Stack is underflow.");
	} else {
		x = arr[top];
		top = top - 1;
		printf("Popped value = %d.", x);
	}
}
					
void display() {
	if (top == -1) {
		printf("Stack is empty.");
	} else {
		printf("Elements are : ");
		int index;
		for (index = top; index >= 0; index--) {
			printf("%d ", arr[index]);
		}
	}
}
